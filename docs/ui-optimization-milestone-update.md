```typescript
// 错误分类和用户友好消息示例
const getUserFriendlyMessage = (errorInfo: ErrorInfo): string => {
  switch (errorInfo.type) {
    case 'network':
      return '网络连接出现问题，请检查网络后重试';
    case 'validation':
      return '输入信息有误，请检查后重新提交';
    case 'permission':
      return '权限不足，请联系管理员或重新登录';
    case 'timeout':
      return '操作超时，请稍后重试';
    default:
      return '操作失败，请稍后重试';
  }
};
```
